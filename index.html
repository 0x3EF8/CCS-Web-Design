-- CCS Student Database
SELECT s.name,
       s.program,
       COUNT(c.id) AS courses
  FROM students s
  JOIN enrollments e
    ON s.id = e.student_id
  JOIN courses c
    ON e.course_id = c.id
 WHERE s.year_level = 1
 ORDER BY s.name ASC;
